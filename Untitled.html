dog_names <- read_csv("https://raw.githubusercontent.com/r-classes/2019_2020_ds4dh_hw_2_dplyr_tidyr_ggplot2/master/data/dog_names.csv")
## Parsed with column specification:
## cols(
##   id = col_double(),
##   name = col_character(),
##   sex = col_character(),
##   age = col_character(),
##   contact_state = col_character()
## )
dog_names <- dog_names %>% 
  select(name, sex) %>%
  count(name, sex, sort = TRUE) %>%
  group_by(sex) %>%
  filter(sex != "Unknown") %>%
  slice(1:20)

dog_names %>%
  ggplot(aes(fct_reorder(name, n), n)) +
  geom_col(aes(fill = sex)) +
  coord_flip() +
  facet_wrap(~sex, scales = "free") +
  scale_fill_brewer(palette="Dark2") +
  labs(x = "",
       y = "count",
       caption = "data from https://pudding.cool/2019/10/shelters/")
